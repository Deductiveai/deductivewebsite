<script>
// Counter animation for success metrics
function animateCounters() {
    const counters = document.querySelectorAll('.counter');
    const speed = 200;

    counters.forEach(counter => {
        const animate = () => {
            const value = +counter.getAttribute('data-target');
            const data = +counter.innerText.replace('%', '');

            const time = value / speed;
            if (data < value) {
                counter.innerText = Math.ceil(data + time) + '%';
                setTimeout(animate, 1);
            } else {
                counter.innerText = value + '%';
            }
        }
        animate();
    });
}

// Trigger animation when section comes into view
const observerOptions = {
    threshold: 0.5
};

const observer = new IntersectionObserver((entries) => {
    entries.forEach(entry => {
        if (entry.isIntersecting) {
            animateCounters();
            observer.unobserve(entry.target);
        }
    });
}, observerOptions);

document.addEventListener('DOMContentLoaded', () => {
    const metricsSection = document.querySelector('.bg-gradient-to-br.from-gray-900');
    if (metricsSection) {
        observer.observe(metricsSection);
    }
});
</script>

    green: { bg: "bg-green-100", text: "text-green-600", stat: "text-green-600" },
    blue: { bg: "bg-blue-100", text: "text-blue-600", stat: "text-blue-600" },
    yellow: { bg: "bg-yellow-100", text: "text-yellow-600", stat: "text-yellow-600" },
    purple: { bg: "bg-purple-100", text: "text-purple-600", stat: "text-purple-600" },
    gray: { bg: "bg-gray-100", text: "text-gray-600", stat: "text-gray-600" },
    indigo: { bg: "bg-indigo-100", text: "text-indigo-600", stat: "text-indigo-600" },
    orange: { bg: "bg-orange-100", text: "text-orange-600", stat: "text-orange-600" },
    red: { bg: "bg-red-100", text: "text-red-600", stat: "text-red-600" },
    pink: { bg: "bg-pink-100", text: "text-pink-600", stat: "text-pink-600" },
    cyan: { bg: "bg-cyan-100", text: "text-cyan-600", stat: "text-cyan-600" },
    teal: { bg: "bg-teal-100", text: "text-teal-600", stat: "text-teal-600" }
};

// Initialize the sector showcase
document.addEventListener('DOMContentLoaded', function() {
    const container = document.getElementById('sectors-container');
    const buttons = document.querySelectorAll('.sector-btn');
    
    // Function to create a sector card
    function createSectorCard(sectorKey, data) {
        const colors = colorMap[data.color] || colorMap.blue;
        
        return `
            <div class="sector-card" data-sector="${sectorKey}">
                <div class="flex items-center mb-6">
                    <div class="w-16 h-16 ${colors.bg} rounded-2xl flex items-center justify-center mr-4">
                        <span class="text-2xl">${data.icon}</span>
                    </div>
                    <div>
                        <h3 class="text-xl font-semibold text-gray-900">${data.title}</h3>
                        <p class="${colors.text} font-medium text-sm">${data.subtitle}</p>
                    </div>
                </div>
                
                <p class="text-gray-600 mb-6 leading-relaxed">${data.description}</p>
                
                <div class="space-y-3 mb-6">
                    ${data.useCases.map(useCase => `
                        <div class="use-case-item">
                            <div class="use-case-icon ${colors.text}"></div>
                            <span class="text-sm text-gray-700">${useCase}</span>
                        </div>
                    `).join('')}
                </div>
                
                <div class="grid grid-cols-2 gap-4 pt-4 border-t border-gray-100">
                    ${data.stats.map(stat => `
                        <div class="text-center">
                            <div class="text-2xl font-bold ${colors.stat} mb-1">${stat.value}</div>
                            <div class="text-xs text-gray-600">${stat.label}</div>
                        </div>
                    `).join('')}
                </div>
            </div>
        `;
    }
    
    // Function to render sectors
    function renderSectors(filterSector = 'all') {
        container.innerHTML = '';
        
        Object.entries(sectorData).forEach(([key, data], index) => {
            if (filterSector === 'all') {
                // Show only first 3 sectors on initial load
                if (index < 3) {
                    const card = createSectorCard(key, data);
                    container.insertAdjacentHTML('beforeend', card);
                }
            } else if (filterSector === key) {
                const card = createSectorCard(key, data);
                container.insertAdjacentHTML('beforeend', card);
            }
        });
    }
    
    // Event listeners for sector buttons
    buttons.forEach(button => {
        button.addEventListener('click', function() {
            const sector = this.dataset.sector;
            
            // Update active button
            buttons.forEach(btn => btn.classList.remove('active'));
            this.classList.add('active');
            
            // Render sectors
            renderSectors(sector);
        });
    });
    
    // Initial render
    renderSectors();
});
</script>
