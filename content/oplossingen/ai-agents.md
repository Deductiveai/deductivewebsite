document.addEventListener('DOMContentLoaded', function() {
    // Agent Workflow Animation
    const agentCards = document.querySelectorAll('.agent-card');
    const taskProgressBars = document.querySelectorAll('.task-progress');
    
    // Animate task progress bars
    function animateProgressBars() {
        taskProgressBars.forEach(bar => {
            const currentWidth = parseInt(bar.style.width);
            const increment = Math.random() * 10 - 5; // Random -5 to +5
            const newWidth = Math.max(10, Math.min(100, currentWidth + increment));
            bar.style.width = newWidth + '%';
        });
    }
    
    // Update agent statuses
    function updateAgentStatuses() {
        const statuses = ['Active', 'Processing', 'Thinking', 'Optimizing'];
        const statusClasses = ['bg-green-100 text-green-800', 'bg-blue-100 text-blue-800', 'bg-yellow-100 text-yellow-800', 'bg-purple-100 text-purple-800'];
        
        document.querySelectorAll('.agent-status').forEach(status => {
            const randomIndex = Math.floor(Math.random() * statuses.length);
            status.textContent = statuses[randomIndex];
            status.className = `agent-status ${statusClasses[randomIndex]} px-2 py-1 rounded-full text-xs`;
        });
    }
    
    // Update agent count
    function updateAgentCount() {
        const count = document.getElementById('agent-count');
        const currentCount = parseInt(count.textContent);
        const newCount = Math.max(2, Math.min(5, currentCount + (Math.random() > 0.5 ? 1 : -1)));
        count.textContent = newCount;
    }
    
    // Start animations
    setInterval(animateProgressBars, 3000);
    setInterval(updateAgentStatuses, 8000);
    setInterval(updateAgentCount, 12000);
    
    // Counter Animation
    function animateCounter(counter) {
        const target = parseInt(counter.dataset.target);
        const increment = target / 100;
        let current = 0;
        
        const timer = setInterval(() => {
            current += increment;
            counter.textContent = Math.ceil(current);
            if (current >= target) {
                counter.textContent = target;
                clearInterval(timer);
            }
        }, 20);
    }
    
    // Intersection Observer for counters
    const counters = document.querySelectorAll('.counter');
    const observer = new IntersectionObserver((entries) => {
        entries.forEach(entry => {
            if (entry.isIntersecting) {
                animateCounter(entry.target);
                observer.unobserve(entry.target);
            }
        });
    });
    
    counters.forEach(counter => observer.observe(counter));
    
    // Communication Signal Animation
    const signals = document.querySelectorAll('.communication-signal');
    signals.forEach(signal => {
        signal.style.animationDelay = Math.random() * 2 + 's';
    });
});
</script>
